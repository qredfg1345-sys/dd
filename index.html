function handleLogin(event) { // Prevent the default form submission (prevents page reload) event.preventDefault(); // Retrieve input values from the login form const user = document.getElementById('user').value; const pass = document.getElementById('pass').value; const role = document.getElementById('role').value; // Validation for Admin Access if (user === 'admin' && pass === 'admin' && role === 'admin') { // Redirect to the Admin Dashboard window.location.href = "admin_dashboard.html"; } // Validation for Operational Staff Access else if (user === 'staff' && pass === 'staff' && role === 'staff') { // Redirect to the Staff Dashboard window.location.href = "staff_dashboard.html"; } // Handle invalid credentials else { // Display a security denial alert alert("Security Denial: Invalid credentials"); } }
